API Documentation

Integrate SpeeDefender's powerful URL and email phishing detection into your applications

REST API Reference

SpeeDefender provides a simple API to analyze URLs and emails programmatically. This documentation explains how to use the API endpoints for your applications.

Base URL

All API endpoints are relative to https://your-app-domain.replit.app/api

Endpoints

GET /analyze

Analyzes a URL for phishing indicators and returns a risk assessment.

Parameters

Name Type Required Description
url string Yes The URL to analyze

Example Request

GET /api/analyze?url=https://example.com

Example Response

{
  "success": true,
  "results": {
    "url": "https://example.com",
    "risk_level": "Low",
    "suspicious_indicators": 1,
    "pattern_analysis": {
      "findings": [
        "URL has a secure protocol (https)",
        "Domain length is appropriate (11 characters)",
        "No IP address used in domain",
        "No suspicious keywords in URL"
      ]
    },
    "domain_info": {
      "findings": [
        "Domain is registered with a known registrar",
        "Domain was created more than 1 year ago"
      ],
      "whois_info": {
        "registrar": "Some Registrar, Inc.",
        "creation_date": "2000-01-01 00:00:00",
        "expiration_date": "2030-01-01 00:00:00"
      }
    },
    "reputation": {
      "findings": [
        "Domain has valid DNS records",
        "Domain is not found in known blacklists"
      ]
    },
    "url_patterns": {
      "suspicious_patterns": [
        "Domain registered relatively recently"
      ]
    }
  },
  "product": "SpeeDefender API",
  "version": "1.0"
}

POST /analyze_email

Analyzes an email for phishing indicators and returns a risk assessment.

Request Body

Name Type Required Description
from string Yes The sender's email address
headers object Yes Email headers as key-value pairs
content string Yes The email body content to analyze

Example Request

POST /api/analyze_email
Content-Type: application/json

{
  "from": "service@example-bank.com",
  "headers": {
    "Return-Path": "",
    "Received": "from mail.suspicious-server.net",
    "Subject": "Urgent: Your Account Needs Attention",
    "Date": "Mon, 3 Apr 2025 10:15:30 -0400"
  },
  "content": "Dear Valued Customer, We have detected suspicious activity on your account. Please click the link below to verify your information: https://secure-banking.example-b4nk.com/login"
}

Example Response

{
  "success": true,
  "results": {
    "from": "service@example-bank.com",
    "risk_level": "High",
    "suspicious_indicators": 4,
    "sender_analysis": {
      "findings": [
        "Sender domain appears suspicious",
        "Mismatch between Return-Path and From address",
        "Sender domain created recently"
      ]
    },
    "header_analysis": {
      "findings": [
        "Suspicious mail server in routing path",
        "Header inconsistencies detected"
      ]
    },
    "content_analysis": {
      "findings": [
        "Contains urgent language patterns",
        "Contains suspicious links",
        "Contains sensitive information requests",
        "URL in content leads to suspicious domain"
      ],
      "suspicious_keywords": [
        "urgent",
        "verify",
        "suspicious activity"
      ]
    }
  },
  "product": "SpeeDefender API",
  "version": "1.0"
}

Code Examples

JavaScript - URL Analysis

// Using fetch API for URL analysis
const apiUrl = 'https://your-app-domain.replit.app/api/analyze?url=https://example.com';

fetch(apiUrl)
  .then(response => response.json())
  .then(data => {
    if (data.success) {
      const results = data.results;
      console.log(`SpeeDefender API v${data.version}`);
      console.log(`Risk level: ${results.risk_level}`);
      console.log(`Suspicious indicators: ${results.suspicious_indicators}`);
    } else {
      console.error(`Error: ${data.message}`);
    }
  })
  .catch(error => console.error('API call failed:', error));

Python - URL Analysis

import requests

# URL Analysis Example
api_url = 'https://your-app-domain.replit.app/api/analyze'
params = {'url': 'https://example.com'}

response = requests.get(api_url, params=params)
data = response.json()

if data['success']:
    results = data['results']
    print(f"Using {data['product']} v{data['version']}")
    print(f"Risk level: {results['risk_level']}")
    print(f"Suspicious indicators: {results['suspicious_indicators']}")
else:
    print(f"Error: {data['message']}")

JavaScript - Email Analysis

// Using fetch API for Email Analysis
const apiUrl = 'https://your-app-domain.replit.app/api/analyze_email';
const emailData = {
  from: 'service@example-bank.com',
  headers: {
    'Return-Path': '',
    'Received': 'from mail.suspicious-server.net',
    'Subject': 'Urgent: Your Account Needs Attention',
    'Date': 'Mon, 3 Apr 2025 10:15:30 -0400'
  },
  content: 'Dear Valued Customer, We have detected suspicious activity on your account. Please click the link below to verify your information: https://secure-banking.example-b4nk.com/login'
};

fetch(apiUrl, {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
  },
  body: JSON.stringify(emailData)
})
  .then(response => response.json())
  .then(data => {
    if (data.success) {
      const results = data.results;
      console.log(`SpeeDefender API v${data.version}`);
      console.log(`Risk level: ${results.risk_level}`);
      console.log(`Suspicious indicators: ${results.suspicious_indicators}`);
      console.log(`Suspicious content: ${results.content_analysis.suspicious_keywords.join(', ')}`);
    } else {
      console.error(`Error: ${data.message}`);
    }
  })
  .catch(error => console.error('API call failed:', error));

Python - Email Analysis

import requests
import json

# Email Analysis Example
api_url = 'https://your-app-domain.replit.app/api/analyze_email'
email_data = {
    'from': 'service@example-bank.com',
    'headers': {
        'Return-Path': '',
        'Received': 'from mail.suspicious-server.net',
        'Subject': 'Urgent: Your Account Needs Attention',
        'Date': 'Mon, 3 Apr 2025 10:15:30 -0400'
    },
    'content': 'Dear Valued Customer, We have detected suspicious activity on your account. Please click the link below to verify your information: https://secure-banking.example-b4nk.com/login'
}

headers = {'Content-Type': 'application/json'}
response = requests.post(api_url, data=json.dumps(email_data), headers=headers)
data = response.json()

if data['success']:
    results = data['results']
    print(f"Using {data['product']} v{data['version']}")
    print(f"Risk level: {results['risk_level']}")
    print(f"Suspicious indicators: {results['suspicious_indicators']}")
    print(f"Suspicious keywords: {', '.join(results['content_analysis']['suspicious_keywords'])}")
else:
    print(f"Error: {data['message']}")

Note

This API is for demonstration purposes. In a production environment, consider implementing rate limiting and API key authentication.
